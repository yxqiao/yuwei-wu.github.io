        :type employees: List[Employee]
        :type id: int

        while queue:
            q = queue.popleft()
            value += q.importance
            if q.subordinates:

                    queue.append(_dict[i])

class MyQueue(object):

    def __init__(self):
        """
        Initialize your data structure here.
        """
        self.stack = []
        

    def push(self, x):
        """
        Push element x to the back of queue.
        :type x: int
        :rtype: None
        """
        self.stack.append(x)
        

    def pop(self):
        """
        Removes the element from in front of queue and returns that element.
        :rtype: int
        """
        a = self.stack[0]
        del self.stack[0]
        return a

    def peek(self):
        """
        Get the front element.
        :rtype: int
        """
        return self.stack[0]
        

    def empty(self):
        """
        Returns whether the queue is empty.
        :rtype: bool
        """
        if not self.stack: return True
        else: return False

class Solution(object):
    def findMaxConsecutiveOnes(self, nums):
        """
        :type nums: List[int]
        :rtype: int
        """
        i = 0
        ans = 0
        while i<len(nums):
            if nums[i] == 1:
                j = i+1
                while j<len(nums) and nums[j] == 1: j+=1            
                ans = max(ans, j-i)
                i = j+1
            else:
                i += 1        
        return ans

### 11 993. Cousins in Binary Tree
```python
# Definition for a binary tree node.
# class TreeNode(object):
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution(object):
    def isCousins(self, root, x, y):
        """
        :type root: TreeNode
        :type x: int
        :type y: int
        :rtype: bool
        """
        depth1,pa1 = self.getLevelUtil(root, x, 0, root)
        depth2,pa2 = self.getLevelUtil(root, y, 0, root)
        return depth1 == depth2 and pa1 != pa2
    
    def getLevelUtil(self, node, data, level, parent): 
        if not node : return 0
        if node.val == data: return level,parent
        return self.getLevelUtil(node.right, data, level + 1, node) or self.getLevelUtil(node.left, data, level + 1, node)
```

class Node(object):
    def __init__(self):
        
        self.children = collections.defaultdict(Node)
        self.isword = False

class Trie(object):

    def __init__(self):
        """
        Initialize your data structure here.
        """
        self.root = Node()

        

    def insert(self, word):
        """
        Inserts a word into the trie.
        :type word: str
        :rtype: None
        """
        current = self.root
        for w in word:
            current = current.children[w]
        current.isword = True       

    def search(self, word):
        """
        Returns if the word is in the trie.
        :type word: str
        :rtype: bool
        """
        current = self.root
        for w in word:
            current = current.children.get(w)
            if current == None:
                return False
        return current.isword

    def startsWith(self, prefix):
        """
        Returns if there is any word in the trie that starts with the given prefix.
        :type prefix: str
        :rtype: bool
        """
        current = self.root
        for w in prefix:
            current = current.children.get(w)
            if current == None:
                return False
        return True

class WordDictionary(object):

    def addWord(self, word):
        """
        Adds a word into the data structure.
        :type word: str
        :rtype: None
        """
        current = self.root
        for w in word:
            current = current.child[w]
        current.isWord = True        

    def search(self, word):
        """
        Returns if the word is in the data structure. A word could contain the dot character '.' to represent any one letter.
        :type word: str
        :rtype: bool
        """
        return self.match(word, 0, self.root)
    
    
    def match(self, word, index, root):
        if root == None:
            return False
        if index == len(word):
            return root.isWord
        if word[index] !='.':
            return root != None and self.match(word,index+1, root.child.get(word[index]))
        else:
            for child in root.child.values():
                if self.match(word,index+1,child):
                    return True
        
        return False
