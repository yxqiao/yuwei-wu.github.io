        :type k: int
        :rtype: int

        res = 0
        i = 0
        j = 1
        while j < len(nums) and i < len(nums):
            if nums[j]- nums[i] < k:
                j += 1
            elif nums[j]- nums[i] == k:
                res += 1

        for num in set(nums):
            if k > 0 and num + k in counter:
                answer += 1
            if k == 0 and counter[num] > 1:
                answer += 1
        return answer
