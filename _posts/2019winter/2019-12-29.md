```python
class Solution(object):
    def countPalindromicSubsequences(self, S):
        """
        :type S: str
        :rtype: int
        """
        if not S: return 0
        n = len(S)
        dp = [[0]*n for _ in range(n)]
        for i in range(n): dp[i][i] = 1
        
        ans = []
        for size in range(2, n+1):
            for i in range(n-size+1):
                j = i+size-1
                if S[i] == S[j]:
                    dp[i][j] = dp[i+1][j-1]*2
                    l = i+1
                    r = j-1
                    while l<=r and S[l] != S[i]: l+=1
                    while l<=r and S[r] != S[i]: r-=1
                    if l>r: dp[i][j] += 2
                    elif l==r: dp[i][j] +=1
                    else: dp[i][j] -= dp[l+1][r-1]
                else: dp[i][j] = dp[i+1][j] + dp[i][j-1] - dp[i+1][j-1] 
                dp[i][j] %= 1000000007
        return  dp[0][n-1]
```

### 3. 31. Next Permutation

切片弄不清楚的时候写个reverse函数就好了
```python
class Solution(object):
    def nextPermutation(self, nums):
        """
        :type nums: List[int]
        :rtype: None Do not return anything, modify nums in-place instead.
        """
        n = len(nums)
        i = n - 1
        while i > 0 and nums[i] <= nums[i - 1]:
            i -= 1
        self.reverse(nums, i, n - 1)
        
        if i > 0:
            for j in range(i, n):
                if nums[j] > nums[i-1]:
                    self.swap(nums, i-1, j)
                    break
        
    def reverse(self, nums, i, j):
        for k in range(i, (i + j) / 2 + 1):
            self.swap(nums, k, i + j - k)

    def swap(self, nums, i, j):
        nums[i], nums[j] = nums[j], nums[i]
```

```python
class Solution(object):
    def findRedundantDirectedConnection(self, edges):
        """
        :type edges: List[List[int]]
        :rtype: List[int]
        """
        graph = collections.defaultdict(list)
        nodeSet = set()
        nodeParent = collections.Counter()
        
        #step 1: fill the grah, nodeset and nodeparent
        for u, v in edges:
            graph[u].append(v)
            nodeParent[v] += 1
            if u not in nodeSet:
                nodeSet.add(u)
            if v not in nodeSet:
                nodeSet.add(v)
         
        #step 2: start searching
        visited = set()
        queue = collections.deque()        
        roots = nodeSet - set(nodeParent.keys())  # do not has the parent
        
        for idx in range(len(edges)-1, -1, -1): # because we need the last answer
            u = edges[idx][0]
            v = edges[idx][1]
            
            if len(roots) == 0:
                if nodeParent[v] == 1:
                    ns = v
                else:
                    continue
            elif len(roots) == 1:
                if nodeParent[v] > 1:
                    ns = list(roots)[0]
                else:
                    continue
                        
            graph[u].remove(v)                
            visited.clear()
            queue.clear()
            queue.append(ns)
            visited.add(ns)
            flag = 1

            while queue:
                node = queue.popleft()

                for nextNode in graph[node]:    
                    if nextNode not in visited:
                        queue.append(nextNode)
                        visited.add(nextNode)
                    else:
                        flag = 0
                        break                        
            graph[u].append(v)
            
            
            if len(visited) != len(nodeSet) or flag == 0:  continue      
                
            return [u, v]


```

```python
                for j in graph[i]:
                    nodedict[i].add(j)
                
        total_path = []
        def dfs(path, s, e):            
            if s==e:
                total_path.append(path[::])
                return True            
            
            for i in nodedict[s]:
                path.append(i)
                dfs(path, i, e)
                path.pop()
                
        dfs([0],0,len(graph)-1)
        return total_path
```
